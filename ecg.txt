fs = 500;
t = 0:1/fs:2;
P_wave = 0.1 * sin(2 * pi * 5 * t);
QRS_wave = 1.5 * exp( - ((t - 0.5) / 0.03).^2);
T_wave = 0.3 * sin(2 * pi * 1.5 * (t - 0.6));
ECG_signal = P_wave + QRS_wave + T_wave;
plot(t, ECG_signal);
title('Synthetic ECG Signal');
xlabel('Time (s)');
ylabel('Amplitude');
